HTML Example 

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>My Website</title>
</head>
<body>
    <header>
        <h1>Welcome to My Site</h1>
        <nav>
            <ul>
                <li><a href="#home">Home</a></li>
                <li><a href="#about">About</a></li>
            </ul>
        </nav>
    </header>
    
    <main>
        <section id="home">
            <h2>About Me</h2>
            <p>I'm learning web development!</p>
            <img src="photo.jpg" alt="My photo">
        </section>
    </main>
    
    <footer>
        <p>© 2024 My Website</p>
    </footer>
</body>
</html>
